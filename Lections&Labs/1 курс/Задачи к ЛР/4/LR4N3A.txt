#include <iostream>

{
    setlocale (LC_ALL, "");
    int n,i,k,p,j,chet,i1;

    cin >> n;

    int* array1 = new int [n];  //заводим массив для первоначального массива который будем считывать
    for(i = 0; i < n; i++)            //заполняем его
        { 
        cin >> array1[i];
        }

    int* array2 = new int [n];     //массив в котором будут хранится элементы без повтора
    for(i = 0; i < n; i++)
        array2[i] = 0;

    for(i = 0; i < n; i++)            //берем каждый элемент массива и проверяем был ли он до него в массиве, если его не было, тогда записываем его в наш новый массив, если был, то пропускаем.
    {
        for(j = 0; j < i; j++)
        {
            if(array1[i] == array2[j])
            k = 1;               //счетчик повтора элемента
        }
 
        if(k == 0)
            array2[i] = array1[i];
        k=0;
    }
 
    for(i = 0; i < n; i++)
    {
        if(array2[i] != 0)
        cout << array1[i] << " ";
    }

}
