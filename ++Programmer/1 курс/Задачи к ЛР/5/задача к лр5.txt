#include <iostream>
#include <string>
#include <stdlib.h>
#include <stdio.h>
using namespace std;

struct CITIES 
    {
        string city;
        int year;
        int citizens;
    };
int main()
{
    setlocale(LC_ALL, "");
    int sumcitizens, maxcitizens; // суммарное кол-во жителей в городах; максимальное кол-во жителей в городе
    string mingorod, maxgorod, maxprirost; // город с наименьшим населением; город с наибольшим населением; город с самым большим приростом населения за последний год

    int size, i;
    cout << "Введите количество городов: ";
    cin >> size;
    CITIES *c = new CITIES[size];
    for (i = 0; i < size; i++)    
    {
        cout << "Город: "; cin >> c[i].city;
        cout << "Год: "; cin >> c[i].year;
        cout << "Жители: "; cin >> c[i].citizens;
        sumcitizens += c[i].citizens; // Сумма всех жителей
    }
    maxcitizens = c[0].citizens;
    maxgorod = c[0].city;
    for (i = 0; i < size; i++) //Цикл поиска самого большого города+наибольшее население
    {
        if (c[i].citizens > maxcitizens)
        {
            maxcitizens = c[i].citizens;
            maxgorod = c[i].city;
        }
    }
    int mingorod1 = c[0].citizens;
    mingorod = c[0].city;
    for (i = 0; i < size; i++){
     //Цикл поиска самого маленького города
        if (c[i].citizens < mingorod1){
        mingorod1 = c[i].citizens;
        mingorod = c[i].city;}
    }
    cout << "суммарное кол-во жителей в городах: " << sumcitizens << endl;
    cout << "максимальное кол-во жителей в городе: " << maxcitizens << endl;
    cout << "Город с наибольшим населением: " << maxgorod << endl;
    cout << "Город с наименьшим населением: " << mingorod << endl;
    cout << "Город с самым большим приростом населения за последний год: " << maxprirost << endl;
    system("pause");
    return 0;
}
